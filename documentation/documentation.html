Zlatan1.RandomInt(0, 10); // Return: 4

Zlatan1.RandomInt(0, 10); // Return: 0

RandomChoiceArray function RandomChoiceArray<T>(const arr: array of T): T;

Função para retornar um valor aleatório de um vetor especificado. 

var
    _intArray: array of Integer;
    _doubleArray: array of Double;
begin
    // Definindo um vetor de Integer (3)
    SetLength(_intArray, 3);
    _intArray[0] := 1;
    _intArray[1] := 2;
    _intArray[2] := 3;

    // Definindo um vetor de Double (3)
    SetLength(_doubleArray, 3);
    _doubleArray[0] := 1.1;
    _doubleArray[1] := 2.2;
    _doubleArray[2] := 3.3;

    Zlatan1.RandomChoiceArray(_intArray);   // Return: 3
    Zlatan1.RandomChoiceArray(_doubleArray); // Return: 2.2
end;
